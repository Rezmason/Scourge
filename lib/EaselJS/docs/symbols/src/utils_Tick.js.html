 32 // constructor:
 33 	/**
 34 	* The Tick class uses a static interface (ex. Tick.getPaused()) and should not be instantiated.
 35 	* @class Provides a centralized tick or heartbeat. Listeners can subscribe to the tick, and identify whether they are pausable or not.
 36 	**/
 37 	function Tick() {
 38 		throw "Tick cannot be instantiated.";
 39 	}
 40 	
 41 // private static properties:
 42 	/** @private **/
 43 	Tick._listeners = null;
 44 	/** @private **/
 45 	Tick._pauseable = null;
 46 	/** @private **/
 47 	Tick._paused = false;
 48 	/** @private **/
 49 	Tick._inited = false;
 50 	/** @private **/
 51 	Tick._startTime = 0;
 52 	/** @private **/
 53 	Tick._pausedTime=0;
 54 	/** @private **/
 55 	Tick._ticks = 0;
 56 	/** @private **/
 57 	Tick._pausedTicks = 0;
 58 	/** @private **/
 59 	Tick._interval = 50; // READ-ONLY
 60 	/** @private **/
 61 	Tick._intervalID = null;
 62 	/** @private **/
 63 	Tick._lastTime = 0;
 64 	
 65 // public static methods:
 66 	/**
 67 	* Adds a listener to the tick. The listener object must expose a .tick() method, which will be called once each tick. The exposed tick method can optionally accept a single parameter, which will include the elapsed time between the previous tick and the current one.
 68 	* @param o The object to add as a listener.
 69 	* @param pausable If false, the listener will continue to have tick called even when Tick is paused via Tick.pause(). Default is false.
 70 	* @static
 71 	**/
 72 	Tick.addListener = function(o, pauseable) {
 73 		if (!Tick._inited) {
 74 			Tick._inited = true;
 75 			Tick.removeAllListeners();
 76 			Tick.setInterval(Tick._interval);
 77 		}
 78 		
 79 		var index = Tick._listeners.indexOf(o);
 80 		if (index != -1) {
 81 			Tick._pauseable[index] = pauseable;
 82 			return;
 83 		}
 84 		
 85 		Tick._pauseable[Tick._listeners.length] = pauseable;
 86 		Tick._listeners.push(o);
 87 	}
 88 	
 89 	/**
 90 	* Removes the specified listener.
 91 	* @param o The listener to remove.
 92 	* @static
 93 	**/
 94 	Tick.removeListener = function(o) {
 95 		if (Tick._listeners == null) { return; }
 96 		var index = Tick._listeners.indexOf(o);
 97 		if (index != -1) {
 98 			Tick._listeners.splice(index,1);
 99 			Tick._pauseable.splice(index,1);
100 		}
101 	}
102 	
103 	/**
104 	* Removes all listeners.
105 	* @static
106 	**/
107 	Tick.removeAllListeners = function() {
108 		Tick._listeners = [];
109 		Tick._pauseable = [];
110 	}
111 	
112 	/**
113 	* Sets the time (in milliseconds) between ticks. Default is 50 (20 FPS).
114 	* @param interval Time in milliseconds between ticks.
115 	* @static
116 	**/
117 	Tick.setInterval = function(interval) {
118 		if (Tick._intervalID != null) { clearInterval(Tick._intervalID); }
119 		Tick._lastTime = Tick._getTime();
120 		Tick._interval = interval;
121 		Tick._intervalID = setInterval(Tick._tick, interval);
122 	}
123 	
124 	/**
125 	* Returns the current time between ticks, as set with setInterval.
126 	* @static
127 	**/
128 	Tick.getInterval = function() {
129 		return Tick._interval;
130 	}
131 	
132 	/**
133 	* Returns the frame rate in frames per second (FPS). For example, with an interval of 40, getFPS() will return 25 (1000ms per second divided by 40 ms per tick = 25fps).
134 	* @static
135 	**/
136 	Tick.getFPS = function() {
137 		return 1000/Tick._interval;
138 	}
139 	
140 	/**
141 	* While Tick is paused, pausable listeners are not ticked. See addListener for more information.
142 	* @param value Indicates whether to pause (true) or unpause (false) Tick.
143 	* @static
144 	**/
145 	Tick.setPaused = function(value) {
146 		Tick._paused = value;
147 	}
148 	
149 	/**
150 	* Returns a boolean indicating whether Tick is currently paused, as set with setPaused.
151 	* @static
152 	**/
153 	Tick.getPaused = function() {
154 		return Tick._paused;
155 	}
156 	
157 	/**
158 	* Returns the number of milliseconds that have elapsed since Tick was loaded. For example, you could use this in a time synchronized animation to determine the exact amount of time that has elapsed.
159 	* @param pauseable Indicates whether to return the elapsed time for pauseable, or unpauseable listeners. If true, time that elapsed while Tick was paused is not included.
160 	* @static
161 	**/
162 	Tick.getTime = function(pauseable) {
163 		return Tick._getTime() - Tick._startTime - (pauseable ? Tick._pausedTime : 0);
164 	}
165 	
166 	/**
167 	* Returns the number of ticks that have elapsed while Tick was active.
168 	* @param pauseable Indicates whether to return the elapsed ticks for pauseable, or unpauseable listeners.
169 	* @static
170 	**/
171 	Tick.getTicks = function(pauseable) {
172 		return  Tick._ticks - (pauseable ?Tick._pausedTicks : 0);
173 	}

175 // private static methods:
176 	/** @private **/
177 	Tick._tick = function() {
178 		Tick._ticks++;
179 		
180 		var time = Tick.getTime(false);
181 		var elapsedTime = time-Tick._lastTime;
182 		var paused = Tick._paused;
183 		
184 		if (paused) {
185 			Tick._pausedTicks++;
186 			Tick._pausedTime += elapsedTime;
187 		}
188 		Tick._lastTime = time;
189 		
190 		var pauseable = Tick._pauseable;
191 		var listeners = Tick._listeners;
192 		
193 		var l = listeners ? listeners.length : 0;
194 		for (var i=0; i<l; i++) {
195 			var p = pauseable[i];
196 			var listener = listeners[i];
197 			if (listener == null || (paused && p) || listener.tick == null) { continue; }
198 			listener.tick(elapsedTime);
199 		}
200 	}
201 	
202 	/** @private **/
203 	Tick._getTime = function() {
204 		new Date().getTime();
205 	}
206 	Tick._startTime = Tick._getTime();
