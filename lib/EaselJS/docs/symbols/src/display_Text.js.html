 32 /**
 33 * Constructs a new Text instance.
 34 * @param text Optional. The text to display.
 35 * @param font Optional. The font style to use. Any valid value for the CSS font attribute is acceptable (ex. "36px bold Arial").
 36 * @param color Optional. The color to draw the text in. Any valid value for the CSS color attribute is acceptable (ex. "#F00").
 37 * @class Allows you to display a single line of dynamic text (not user editable) in the display list. Note that as an alternative to Text, you can position HTML text above or below the canvas relative to items in the display list using the localToGlobal() method.
 38 * @augments DisplayObject
 39 **/
 40 function Text(text, font, color) {
 41   this.init(text, font, color);
 42 this.prototype = new DisplayObject();
 43 
 44 /** @private **/
 45 var canvas = document.createElement("canvas");
 46 Text._workingContext = canvas.getContext("2d");
 47 
 48 // public properties:
 49 	/** The text to display. **/
 50 	this.text = "";
 51 	/** The font style to use. Any valid value for the CSS font attribute is acceptable (ex. "bold 36px Arial"). **/
 52 	this.font = null;
 53 	/** The color to draw the text in. Any valid value for the CSS color attribute is acceptable (ex. "#F00"). **/
 54 	this.color = null;
 55 	/** The horizontal text alignment. Any of start, end, left, right, and center. For detailed information view the <a href="http://www.whatwg.org/specs/web-apps/current-work/multipage/the-canvas-element.html#text-0">whatwg spec</a>. **/
 56 	this.textAlign = null;
 57 	/** The vertical alignment point on the font. Any of top, hanging, middle, alphabetic, ideographic, or bottom. For detailed information view the <a href="http://www.whatwg.org/specs/web-apps/current-work/multipage/the-canvas-element.html#text-0">whatwg spec</a>. **/
 58 	this.textBaseline = null;
 59 	/** The maximum width to draw the text. If maxWidth is specifiied (not null), the text will be condensed or shrunk to make it fit in this width. For detailed information view the <a href="http://www.whatwg.org/specs/web-apps/current-work/multipage/the-canvas-element.html#text-0">whatwg spec</a>. **/
 60 	this.maxWidth = null;
 61 	/** If true, the text will be drawn as a stroke (outline). If false, the text will be drawn as a fill. **/
 62 	this.outline = false;
 63 	
 64 // constructor:
 65 	/** @private **/
 66 	this._init = this.init;
 67 	/** @private **/
 68 	this.init = function(text, font, color) {
 69 		this._init();
 70 		this.text = text;
 71 		this.font = font;
 72 		this.color = color ? color : "#000";
 73 	}
 74 	
 75 // public methods:
 76 	this._draw = this.draw;
 77 	this.draw = function(ctx,ignoreCache) {
 78 		if (!this._draw(ctx,ignoreCache) || this.text == null || this.text.length == 0) { return false; }
 79 		
 80 		if (this.outline) { ctx.strokeStyle = this.color; }
 81 		else { ctx.fillStyle = this.color; }
 82 		ctx.font = this.font;
 83 		ctx.textAlign = this.textAlign ? this.textAlign : "start";
 84 		ctx.textBaseline = this.textBaseline ? this.textBaseline : "alphabetic";
 85 		if (this.outline) { ctx.strokeText(this.text, 0, 0, this.maxWidth); }
 86 		else { ctx.fillText(this.text, 0, 0, this.maxWidth); }
 87 	}
 88 	
 89 	/**
 90 	* Returns the measured, untransformed width of the text.
 91 	**/
 92 	this.getMeasuredWidth = function() {
 93 		var ctx = Text._workingContext;
 94 		ctx.font = this.font;
 95 		ctx.textAlign = this.textAlign ? this.textAlign : "start";
 96 		ctx.textBaseline = this.textBaseline ? this.textBaseline : "alphabetic";
 97 		return ctx.measureText(this.text).width;
 98 	}
 99 	
100 	this.clone = function() {
101 		var o = new Text(this.text, this.font, this.color);
102 		this.cloneProps(o);
103 		return o;
104 	}
105 		
106 	this.toString = function() {
107 		return "[Text (text="+  (this.text.length > 20 ? this.text.substr(0,17)+"..." : this.text) +")]";
108 	}
109 	
110 // private methods:
111 	
112 	/** @private **/
113 	this._cloneProps = this.cloneProps;
114 	/** @private **/
115 	this.cloneProps = function(o) {
116 		this._cloneProps(o);
117 		o.textAlign = this.textAlign;
118 		o.textBaseline = this.textBaseline;
119 		o.maxWidth = this.maxWidth;
120 		o.outline = this.outline;
121 	}
122 }
