 32 /**
 33 * DisplayObject is an abstract class that should not be constructed directly. Instead construct subclasses such as Sprite, Bitmap, and Shape.
 34 * @class DisplayObject is the base class for all display classes in the CanvasDisplay library. It defines the core properties and methods that are shared between all display objects. It should not be instantiated directly.
 35 **/
 36 function DisplayObject() {
 37   this.init();
 38 this.prototype;
 39 
 40 // public properties:
 41 	/** The alpha (transparency) for this display object. 0 is fully transparent, 1 is fully opaque. **/
 42 	this.alpha = 1;
 43 	/** If a cache is active, this returns the canvas that holds the cached version of this display object. See cache() for more information. READ-ONLY. **/
 44 	this.cacheCanvas = null;
 45 	/** Unique ID for this display object. Makes display objects easier for some uses. **/
 46 	this.id = -1;
 47 	/** Indicates whether to include this object when running Stage.getObjectsUnderPoint(). Setting this to true for Sprites will cause the Sprite to be returned (not its children) regardless of whether it's mouseChildren property is true. **/
 48 	this.mouseEnabled = false;
 49 	/** An optional name for this display object. Included in toString(). Useful for debugging. **/
 50 	this.name = null;
 51 	/** A reference to the Sprite or Stage object that contains this display object, or null if it has not been added to one. READ-ONLY. **/
 52 	this.parent = null;
 53 	/** The x offset for this display object's registration point. For example, to make a 100x100px Bitmap rotate around it's center, you would set regX and regY to 50. **/
 54 	this.regX = 0;
 55 	/** The y offset for this display object's registration point. For example, to make a 100x100px Bitmap rotate around it's center, you would set regX and regY to 50. **/
 56 	this.regY = 0;
 57 	/** The rotation in degrees for this display object. **/
 58 	this.rotation = 0;
 59 	/** The factor to stretch this display object horizontally. For example, setting scaleX to 2 will stretch the display object to twice it's nominal width. **/
 60 	this.scaleX = 1;
 61 	/** The factor to stretch this display object vertically. For example, setting scaleY to 0.5 will stretch the display object to half it's nominal height. **/
 62 	this.scaleY = 1;
 63 	/** A shadow object that defines the shadow to render on this display object. Set to null to remove a shadow. Note that nested shadows can result in unexpected behaviour (ex. if both a child and a parent have a shadow set). **/
 64 	this.shadow = null;
 65 	/** Indicates whether this display object should be rendered to the canvas and included when running Stage.getObjectsUnderPoint(). **/
 66 	this.visible = true;
 67 	/** The x (horizontal) position of the display object, relative to its parent. **/
 68 	this.x = 0;
 69 	/** The y (vertical) position of the display object, relative to its parent. **/
 70 	this.y = 0;
 71 	
 72 // private properties:
 73 	/** @private **/
 74 	this._cacheOffsetX = 0;
 75 	/** @private **/
 76 	this._cacheOffsetY = 0;
 77 	/** @private **/
 78 	this._cacheDraw = false;
 79 	/** @private **/
 80 	this._activeContext = null;
 81 	/** @private **/
 82 	this._restoreContext = false;
 83 	/** @private **/
 84 	this._revertShadow = false;
 85 	/** @private **/
 86 	this._revertX = 0;
 87 	/** @private **/
 88 	this._revertY = 0;
 89 	/** @private **/
 90 	this._revertAlpha = 1;
 91 	
 92 // constructor:
 93 	// separated so it can be easily addressed in subclasses:
 94 	/** @private **/
 95 	this.init = function() {
 96 		this.id = UID.get();
 97 		this.children = [];
 98 	}
 99 	
100 // public methods:
101 	/**
102 	* NOTE: This method is mainly for internal use, though it may be useful for advanced developers extending the capabilities of the CanvasDisplay library.
103 	* Updates the specified context based on this display object's properties.
104 	* @param ctx The canvas 2D context object to update.
105 	* @param ignoreShadows Indicates whether the shadow property should be applied. Passing false will ignore the shadow, resulting in faster rendering for uses like hit testing.
106 	**/
107 	this.updateContext = function(ctx,ignoreShadows) {
108 		if (this._activeContext != null) { throw "ERROR: updateContext called twice without calling revertContext. ["+this.name+"]"; } // GDS: temp for catching errors.
109 		if (this.visible != true || ctx == null || this.alpha <= 0) { return false; }
110 		// apply context changes:
111 		this._activeContext = ctx;
112 		
113 		if (this._restoreContext = (this.rotation%360 || this.scaleX != 1 || this.scaleY != 1)) {
114 			ctx.save();
115 			// GDS: might be worth benchmarking implicit vs explicit boolean tests here:
116 			if (this.x || this.y) { ctx.translate(this.x, this.y); }
117 			if (this.rotation%360) { ctx.rotate(this.rotation%360/180*Math.PI); }
118 			if (this.scaleX != 1 || this.scaleY != 1) { ctx.scale(this.scaleX, this.scaleY); }
119 			if (this.regX || this.regY) { ctx.translate(-this.regX, -this.regY); }
120 		} else {
121 			ctx.translate(-(this._restoreX = -this.x+this.regX), -(this._restoreY = -this.y+this.regY));
122 		}
123 		
124 		this._revertAlpha = ctx.globalAlpha;
125 		ctx.globalAlpha *= this.alpha;
126 		if (this._revertShadow = (this.shadow && !ignoreShadows)) {
127 			this.applyShadow(ctx, this.shadow);
128 		}
129 	}
130 	
131 	/**
132 	* NOTE: This method is mainly for internal use, though it may be useful for advanced developers extending the capabilities of the CanvasDisplay library.
133 	* Draws the display object into the specified context if it is visible.
134 	* @param ctx The canvas 2D context object to draw into.
135 	* @param ignoreCache Indicates whether the draw operation should ignore any current cache. For example, used for drawing the cache (to prevent it from simply drawing an existing cache back into itself).
136 	**/
137 	this.draw = function(ctx,ignoreCache) {
138 		// can't use _activeContext because sometimes we need to draw without an updateContext being called (ex. caching)
139 		if (this.visible != true || ctx == null || this.alpha <= 0) { return false; }
140 		if (this.cacheCanvas && !ignoreCache) {
141 			ctx.translate(this._cacheOffsetX,this._cacheOffsetY);
142 			ctx.drawImage(this.cacheCanvas,0,0);
143 			ctx.translate(-this._cacheOffsetX,-this._cacheOffsetY);
144 			return false;
145 		}
146 		return true;
147 	}
148 	
149 	/**
150 	* NOTE: This method is mainly for internal use, though it may be useful for advanced developers extending the capabilities of the CanvasDisplay library.
151 	* Reverts the last context that was updated with updateContext(), restoring it to the state it was in prior to the update.
152 	**/
153 	this.revertContext = function() {
154 		if (this._activeContext == null) { return; }
155 		this._activeContext.globalAlpha = this._revertAlpha;
156 		if (this._revertShadow) {
157 			// GDS: instead, we could save out the shadow properties in update, and restore them here.
158 			this.applyShadow(this._activeContext, Shadow.identity);
159 		}
160 		if (this._restoreContext) { this._activeContext.restore(); }
161 		else { this._activeContext.translate(this._restoreX, this._restoreY); }
162 		this._activeContext = null
163 	}
164 	
165 	/**
166 	* Draws the display object into a new canvas, which is then used for subsequent draws. For complex content that does not change frequently (ex. a Sprite with many children that do not move, or a complex vector Shape), this can provide for much faster rendering because the content does not need to be rerendered each tick. The cached display object can be moved, rotated, faded, etc freely, however if it's content changes, you must manually update the cache by calling cache() again. Do not call uncache before the subsequent cache call. You must specify the cache area via the x, y, w, and h parameters. This defines the rectangle that will be rendered and cached using this display object's coordinates. For example if you defined a Shape that drew a circle at 0,0 with a radius of 25, you could call myShape.cache(-25,-25,50,50) to cache the full shape.
167 	* @param x
168 	* @param y
169 	* @param width
170 	* @param height
171 	**/
172 	this.cache = function(x, y, w, h) {
173 		// draw to canvas.
174 		var ctx;
175 		if (this.cacheCanvas == null) {
176 			this.cacheCanvas = document.createElement("canvas");
177 			ctx = this.cacheCanvas.getContext("2d");
178 		} else {
179 			ctx = this.cacheCanvas.getContext("2d");
180 			ctx.clearRect(0,0,this.cacheCanvas.width,this.cacheCanvas.height);
181 		}
182 		this.cacheCanvas.width = w;
183 		this.cacheCanvas.height = h;
184 		ctx.translate(-x,-y);
185 		this.draw(ctx,true);
186 		this._cacheOffsetX = x;
187 		this._cacheOffsetY = y;
188 	}
189 	
190 	/**
191 	* Clears the current cache. See cache() for more information.
192 	**/
193 	this.uncache = function() {
194 		this.cacheCanvas = null;
195 		this.cacheOffsetX = this.cacheOffsetY = 0;
196 	}
197 	
198 	/**
199 	* Returns the stage that this display object will be rendered on, or null if it has not been added to one.
200 	**/
201 	this.getStage = function() {
202 		var o = this;
203 		while (o.parent) {
204 			o = o.parent;
205 		}
206 		if (o instanceof Stage) { return o; }
207 		return null;
208 	}
209 	
210 	/**
211 	* Returns a clone of this DisplayObject. Some properties that are specific to this instance's current context are reverted to their defaults (for example .parent).
212 	**/
213 	this.clone = function() {
214 		var o = new DisplayObject();
215 		this.cloneProps(o);
216 		return o;
217 	}
218 	
219 	/**
220 	* Returns a string representation of this object.
221 	**/
222 	this.toString = function() {
223 		return "[DisplayObject (name="+  this.name +")]";
224 	}
225 	
226 // private methods:
227 	// separated so it can be used easily in subclasses:
228 	/** @private **/
229 	this.cloneProps = function(o) {
230 		o.alpha = this.alpha;
231 		o.name = this.name;
232 		o.regX = this.regX;
233 		o.regY = this.regY;
234 		o.rotation = this.rotation;
235 		o.scaleX = this.scaleX;
236 		o.scaleY = this.scaleY;
237 		o.shadow = this.shadow;
238 		o.visible = this.visible;
239 		o.x  = this.x;
240 		o.y = this.y;
241 		o.mouseEnabled = this.mouseEnabled;
242 	}
243 	
244 	/** @private **/
245 	this.applyShadow = function(ctx, shadow) {
246 		ctx.shadowColor = shadow.color;
247 		ctx.shadowOffsetX = shadow.offsetX;
248 		ctx.shadowOffsetY = shadow.offsetY;
249 		ctx.shadowBlur = shadow.blur;
250 	}
251 }
