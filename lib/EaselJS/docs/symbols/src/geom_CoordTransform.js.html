 33 // constructor:
 34 	/**
 35 	* The CoordTransform class uses a static interface and should not be instantiated.
 36 	* @class The CoordTransform class is a collection of methods used to transform coordinates between different display object's coordinate spaces.
 37 	**/
 38 	function CoordTransform() {
 39 		throw "CoordTransform cannot be instantiated"; 
 40 	}
 41 	
 42 // public static methods:
 43 	/**
 44 	* Transforms the specified x and y position from the coordinate space of the source display object
 45 	* to the global (stage) coordinate space. For example, this could be used to position an HTML label
 46 	* over a specific point on a nested display object. Returns a Point instance with x and y properties
 47 	* correlating to the tranformed coordinates on the stage.
 48 	* @param x The x position in the source display object to transform.
 49 	* @param y The y position in the source display object to transform.
 50 	* @param target The source display object.
 51 	* @static
 52 	**/
 53 	CoordTransform.localToGlobal = function(x, y, source) {
 54 		var mtx = CoordTransform.getConcatenatedMatrix(source);
 55 		if (mtx == null) { return null; }
 56 		var mtx2 = new Matrix2D(1,0,0,1,x,y);
 57 		mtx2.concatMatrix(mtx);
 58 		return new Point(mtx2.tx, mtx2.ty);
 59 	}
 60 	
 61 	/**
 62 	* Transforms the specified x and y position from the global (stage) coordinate space to the
 63 	* coordinate space of the target display object. For example, this could be used to determine
 64 	* the current mouse position within a specific display object. Returns a Point instance with x and y properties
 65 	* correlating to the tranformed position in the target's coordinate space.
 66 	* @param x The x position on the stage to transform.
 67 	* @param y The y position on the stage to transform.
 68 	* @param target The target display object.
 69 	* @static
 70 	**/
 71 	CoordTransform.globalToLocal = function(x, y, target) {
 72 		var mtx = CoordTransform.getConcatenatedMatrix(target);
 73 		if (mtx == null) { return null; }
 74 		mtx.invert();
 75 		var mtx2 = new Matrix2D(1,0,0,1,x,y);
 76 		mtx2.concatMatrix(mtx);
 77 		return new Point(mtx2.tx, mtx2.ty);
 78 	}
 79 	
 80 	/**
 81 	* Transforms the specified x and y position from the coordinate space of the source display object to the
 82 	* coordinate space of the target display object. Returns a Point instance with x and y properties
 83 	* correlating to the tranformed position in the target's coordinate space. Effectively the same as calling
 84 	* var pt = localToGlobal(x, y, source); pt = globalToLocal(pt.x, pt.y, target);
 85 	* @param x The x position in the source display object to transform.
 86 	* @param y The y position on the stage to transform.
 87 	* @param source The display object that the original position is relative to.
 88 	* @param target The target display object to which the coordinates will be transformed.
 89 	* @static
 90 	**/
 91 	CoordTransform.localToLocal = function(x, y, source, target) {
 92 		// GDS: this could be optimized to find the nearest shared parent, and pass it in as the goal.
 93 		var pt = CoordTransform.localToGlobal(x, y, source);
 94 		return CoordTransform.globalToLocal(pt.x, pt.y, target);
 95 	}
 96 	
 97 	/**
 98 	* Generates a concatenated Matrix2D object representing the combined tranform of
 99 	* the target display object and all of its parent Containers. This can be used to transform
100 	* positions between coordinate spaces, as with localToGlobal and globalToLocal.
101 	* @param target The target display object to which the coordinates will be transformed.
102 	* @param goal Optional parameter specifying the parent Container to concanate to. If it is not included, or is not a valid parent of the target, then the matrix will be concatenated to the highest level parent (usually the stage).
103 	* @static
104 	**/
105 	CoordTransform.getConcatenatedMatrix = function(target, goal) {
106 		var path = [target];
107 		while (target = target.parent) {
108 			path.push(target);
109 			if (target == goal) { break; }
110 		}
111 		var l = path.length;
112 		var stage = path[l-1];
113 		if (!(stage instanceof Stage)) { return null; }
114 		var mtx = new Matrix2D();
115 		for (var i=0; i<l; i++) {
116 			var target = path[i];
117 			mtx.concatTransform(target.x, target.y, target.scaleX, target.scaleY, target.rotation, target.regX, target.regY);
118 		}
119 		return mtx;
120 	}
