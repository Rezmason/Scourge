 32 /**
 33 * Constructs a new Graphics instance.
 34 * @param instructions Optional. The vector drawing instruction that this Graphics will use. See the instructions property for information.
 35 * @class A Graphics instance encapsulates a set of drawing instructions so they can be rendered to the stage within the current context. Generally, you would use the drawing API
 36 * exposed by Graphics to generate instructions rather than setting them directly. Note that you can use Graphics without any dependency on the Easel framework by calling draw() directly,
 37 * or it can be used with the Shape object to draw vector graphics within the context of an Easel display list.<br/><br/>
 38 * Note that all drawing methods in Graphics return the Graphics instance, so they can be chained together. For example, the following line of code would draw a rectangle with a red stroke and blue fill:<br/>
 39 * myGraphics.beginStroke("#F00").beginFill("#00F").drawRect(20, 20, 100, 50);
 40 * @augments DisplayObject
 41 **/
 42 function Graphics(instructions) {
 43 	this.init(instructions);
 44 this.prototype;

 46 // static public methods:

 48 	/**
 49 	* Returns a CSS compatible color string based on the specified RGB numeric color values in the format "rgba(255,255,255,1.0)", or if alpha is null then in the format "rgb(255,255,255)". For example,
 50 	* Graphics.getRGB(50,100,150,0.5) will return "rgba(50,100,150,0.5)".
 51 	* @param r The red component for the color, between 0 and 0xFF (255).
 52 	* @param g The green component for the color, between 0 and 0xFF (255).
 53 	* @param b The blue component for the color, between 0 and 0xFF (255).
 54 	* @param alpha Optional. The alpha component for the color where 0 is fully transparent and 1 is fully opaque.
 55 	* @static
 56 	**/
 57 	Graphics.getRGB = function(r, g, b, alpha) {
 58 		if (alpha == null) {
 59 			return "rgb("+r+","+g+","+b+")";
 60 		} else {
 61 			return "rgba("+r+","+g+","+b+","+alpha+")";
 62 		}
 63 	}
 64 	
 65 	/**
 66 	* Returns a CSS compatible color string based on the specified HSL numeric color values in the format "hsla(360,100,100,1.0)", or if alpha is null then in the format "hsl(360,100,100)". For example,
 67 	* Graphics.getHSL(150,100,70) will return "hsl(150,100,70)".
 68 	* @param hue The hue component for the color, between 0 and 360.
 69 	* @param saturation The saturation component for the color, between 0 and 100.
 70 	* @param lightness The lightness component for the color, between 0 and 100.
 71 	* @param alpha Optional. The alpha component for the color where 0 is fully transparent and 1 is fully opaque.
 72 	* @static
 73 	**/
 74 	Graphics.getHSL = function(hue, saturation, lightness, alpha) {
 75 		if (alpha == null) {
 76 			return "hsl("+(hue%360)+","+saturation+"%,"+lightness+"%)";
 77 		} else {
 78 			return "hsla("+(hue%360)+","+saturation+"%,"+lightness+"%,"+alpha+")";
 79 		}
 80 	}

234 	/**
235 	* Begins a fill with the specified color. This ends the current subpath.
236 	* @param color A CSS compatible color value (ex. "#FF0000" or "rgba(255,0,0,0.5)"). Setting to null will result in no fill.
237 	**/
238 	this.beginFill = function(color) {
239 		if (this._active) { this._newPath(); }
240 		if (color == null) { this._fillInstructions = null; }
241 		else { this._fillInstructions = "c.fillStyle='"+color+"';"; }
242 		return this;

245 	/**
246 	* Begins a linear gradient fill defined by the line (x0,y0) to (x1,y1). This ends the current subpath. For example, the following code defines a black to white vertical gradient ranging from 20px to 120px, and draws a square to display it:<br/>
247 	* myGraphics.beginLinearGradientFill(["#000","#FFF"], [0,1], 0, 20, 0, 120).drawRect(20,20,120,120);
248 	* @param colors An array of CSS compatible color values. For example, ["#F00","#00F"] would define a gradient drawing from red to blue.
249 	* @param ratios An array of gradient positions which correspond to the colors. For example, [0.1,0.9] would draw the first color to 10% then interpolating to the second color at 90%.
250 	* @param x0 The position of the first point defining the line that defines the gradient direction and size.
251 	* @param y0 The position of the first point defining the line that defines the gradient direction and size.
252 	* @param x1 The position of the second point defining the line that defines the gradient direction and size.
253 	* @param y1 The position of the second point defining the line that defines the gradient direction and size.
254 	**/
255 	this.beginLinearGradientFill = function(colors, ratios, x0, y0, x1, y1) {
256 		if (this._active) { this._newPath(); }
257 		this._fillInstructions = "o=c.createLinearGradient("+x0+","+y0+","+x1+","+y1+");";
258 		var l = colors.length;
259 		for (var i=0; i<l; i++) {
260 			this._fillInstructions += "o.addColorStop("+ratios[i]+",'"+colors[i]+"');";
261 		}
262 		this._fillInstructions += "c.fillStyle=o;"
263 		return this;
264 	}

311 	/**
312 	* Sets the stroke style for the current subpath. Like all drawing methods, this can be chained, so you can define the stroke style and color in a single line of code like so:
313 	* myGraphics.setStrokeStyle(8,"round").beginStroke("#F00");

321 		this._strokeStyleInstructions = "c.lineWidth="+(thickness != null ? thickness : "1")+";"
322 			+ "c.lineCap='" + (caps ? caps : "butt") +"';"
323 			+ "c.lineJoin='" + (joints ? joints : "miter") +"';"
324 			+ "c.miterLimit="+ (miterLimit ? miterLimit : "10")+";";
325 		return this;
326 	}
