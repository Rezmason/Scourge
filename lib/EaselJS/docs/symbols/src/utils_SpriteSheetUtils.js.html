 42 	/**
 43 	* Builds a new extended sprite sheet based on the specified image adding flipped frames (vertical, horizontal, or both). Flipping elements on the display list by using setting scaleX/scaleY to -1 is quite expensive in most browsers, so this method allows you to incur the cost of flipping once, in advance, without increasinig the load size of your sprite sheets. Returns a generic object with a property "image" containing the new sprite sheet as an Image instance, and "frameData" which contains the new frame data object.
 44 	* @param image The sprite sheet (canvas, image or video) to use as the source material image.
 45 	* @param frameWidth The width of each frame in the sprite sheet.
 46 	* @param frameHeight The height of each frame in the sprite sheet.
 47 	* @param frameData The frameData that defines the frames and sequences in the sprite sheet. See BitmapSequence.frameData for more information.
 48 	* @param flipData A generic object that specifies which frames will be flipped, what to name the flipped result, and how to flip the frames (horizontally, vertically, or both). Each property name indicates the name of a new sequence to create, and should reference an array where the first index is the name of the original sequence to flip, the second index indicates whether to flip it horizontally, the third index indicates whether to flip it vertically, and the fourth indicates what the "next" value for the resulting frame data should be. For example, the following would create a new sequence named "walk_left" consisting of the frames from the original "walk_right" sequence flipped horizontally: {walk_left:["walk_right", true, false]}
 49 	* @static
 50 	**/
 51 	SpriteSheetUtils.flip = function(spriteSheet, flipData) {
 52 		var image = spriteSheet.image;
 53 		var frameData = spriteSheet.frameData;
 54 		var frameWidth = spriteSheet.frameWidth;
 55 		var frameHeight = spriteSheet.frameHeight;
 56 		
 57 		var cols = image.width/frameWidth|0;
 58 		var rows = image.height/frameHeight|0;
 59 		var ttlFrames = cols*rows;
 60 		
 61 		// clone frameData:
 62 		var frData = {};
 63 		for (var n in frameData) {
 64 			data = frameData[n];
 65 			if (data instanceof Array) { data = data.slice(0); }
 66 			frData[n] = data;
 67 		}
 68 		
 69 		// calculate how many new frames we're generating, and build a map:
 70 		var map = [];
 71 		var frCount = 0;
 72 		var i = 0;
 73 		for (var n in flipData) {
 74 			var fd = flipData[n];
 75 			var data = frameData[fd[0]];
 76 			if (data ==  null) { continue; }
 77 			if (data instanceof Array) {
 78 				var start = data[0];
 79 				var end = data[1];
 80 				if (end == null) { end = start; }
 81 			} else {
 82 				start = end = data;
 83 			}
 84 			map[i] = n;
 85 			map[i+1] = start;
 86 			map[i+2] = end;
 87 			frCount += end-start+1;
 88 			i+=4;
 89 		}
 90 		
 91 		// get the canvas ready:
 92 		var canvas = document.createElement("canvas");
 93 		canvas.width = image.width;
 94 		canvas.height = Math.ceil(rows+frCount/cols)*frameHeight;
 95 		var ctx = canvas.getContext("2d");
 96 		ctx.drawImage(image, 0, 0, cols*frameWidth, rows*frameHeight, 0, 0, cols*frameWidth, rows*frameHeight);
 97 		
 98 		// draw the new frames, and update the new frameData:
 99 		var frame = ttlFrames-1;
100 		for (i=0; i<map.length; i+=4) {
101 			n = map[i];
102 			start = map[i+1];
103 			end = map[i+2];
104 			fd = flipData[n];
105 			var flipH = fd[1]?-1:1;
106 			var flipV = fd[2]?-1:1;
107 			var offH = flipH==-1?frameWidth:0;
108 			var offV = flipV==-1?frameHeight:0;
109 			
110 			for (j=start; j<=end; j++) {
111 				frame++;
112 				ctx.save();
113 				ctx.translate((frame%cols)*frameWidth+offH, (frame/cols|0)*frameHeight+offV);
114 				ctx.scale(flipH, flipV);
115 				ctx.drawImage(image, (j%cols)*frameWidth, (j/cols|0)*frameHeight, frameWidth, frameHeight, 0, 0, frameWidth, frameHeight);
116 				
117 				ctx.restore();
118 			}
119 			frData[n] = [frame-(end-start), frame, fd[3]];
120 			
121 		}
122 		
123 		var img = new Image();
124 		img.src = canvas.toDataURL("image/png");
125 		return new SpriteSheet((img.width > 0)? img : canvas, frameWidth, frameHeight, frData);
126 	}

128 	/**
129 	* Returns a string representing the specified frameData object.
130 	* @param frameData The frame data to output.
131 	* @static
132 	**/
133 	SpriteSheetUtils.frameDataToString = function(frameData) {
134 		var str = "";
135 		var max = 0;
136 		var min = 0;
137 		var count = 0;
138 		for (var n in frameData) {
139 			count++;
140 			data = frameData[n];
141 			if (data instanceof Array) {
142 				var start = data[0];
143 				var end = data[1];
144 				if (end == null) { end = start; }
145 				next = data[2];
146 				if (next == null) { next = n; }
147 			} else {
148 				start = end = data;
149 				next = n;
150 			}
151 			str += "\n\t"+n+", start="+start+", end="+end+", next="+next;
152 			if (next == false) { str += " (stop)"; }
153 			else if (next == n) { str += " (loop)"; }
154 			if (end > max) { max = end; }
155 			if (start < min) { min = start; }
156 		}
157 		str = count+" sequences, min="+min+", max="+max+str;
158 		return str;
159 	}
