 32 /**
 33 * Constructs a Stage object with the specified target canvas.
 34 * @param canvas The canvas the stage will render to.
 35 * @class A stage is the root level Container for a display list. Each time its tick method is called, it will render its display list to its target canvas.
 36 * @augments Container
 37 **/
 38 function Stage(canvas) {
 39   this.init(canvas);
 40 this.prototype = new Container();
 41 
 42 // public properties:
 43 	/** Indicates whether the stage should automatically clear the canvas before each render. You can set this to false to manually control clearing (for generative art, or when pointing multiple stages at the same canvas for example). **/
 44 	this.autoClear = true;
 45 	/** The canvas the stage will render to. Multiple stages can share a single canvas, but you must disable autoClear for all but the first stage that will be ticked (or they will clear each other's render). **/
 46 	this.canvas = null;
 47 	this.mouseX = null;
 48 	this.mouseY = null;
 49 	
 50 	
 51 // private properties:
 52 	/** @private **/
 53 	this._tmpCanvas = null;
 54 	
 55 // constructor:
 56 	/** @private **/
 57 	this.__init = this.init;
 58 	/** @private **/
 59 	this.init = function(canvas) {
 60 		this.__init();
 61 		this.canvas = canvas;
 62 		
 63 		var o = this;
 64 		if (window.addEventListener) {
 65 			window.addEventListener('mousemove', function(e) { o._handleMouseMove(e); }, false); 
 66 		} else if (document.addEventListener) {
 67 			 document.addEventListener('mousemove', function(e) { o._handleMouseMove(e); }, false); 
 68 		} else if (window.attachEvent) {
 69 			window.attachEvent('mousemove', function(e) { o._handleMouseMove(e); }); 
 70 		}
 71 	}
 72 	
 73 // public methods:
 74 	/**
 75 	* Each time tick is called, the stage will render its entire display list to the canvas.
 76 	**/
 77 	this.tick = function() {
 78 		if (this.canvas == null) { return; }
 79 		var ctx = this.canvas.getContext("2d");
 80 		if (this.autoClear) { this.clear(); }
 81 		this.updateContext(ctx);
 82 		this.draw(ctx);
 83 		this.revertContext();
 84 	}
 85 	
 86 	/**
 87 	* Clears the target canvas. Useful if autoClear is set to false.
 88 	**/
 89 	this.clear = function() {
 90 		if (this.canvas == null) { return; }
 91 		this.canvas.getContext("2d").clearRect(0,0,this.canvas.width,this.canvas.height);
 92 	}
 93 	
 94 	/**
 95 	* Returns an array of all display objects under the specified canvas coordinates that are in this stage's display list. This routine ignores any display objects with mouseEnabled set to false (the default) or that are inside containers with mouseChildren set to false (the default). The array will be sorted in order of visual depth, with the top-most display object at index 0. This uses shape based hit detection, and can be an expensive operation to run, so it is best to use it carefully. For example, if testing for objects under the mouse, test on tick (instead of on mousemove), and only if the mouse's position has changed.
 96 	* @param x The x coordinate to test.
 97 	* @param y The y coordinate to test.
 98 	**/
 99 	this.getObjectsUnderPoint = function(x,y) {
100 		var arr = [];
101 		this._getObjectsUnderPoint(x,y,arr);
102 		return arr;
103 	}
104 	
105 	/**
106 	* Similar to getObjectsUnderPoint(), but returns only the top-most display object. This runs significantly faster than getObjectsUnderPoint(), but is still an expensive operation. See getObjectsUnderPoint() for more information.
107 	* @param x The x coordinate to test.
108 	* @param y The y coordinate to test.
109 	**/
110 	this.getObjectUnderPoint = function(x,y) {
111 		return this._getObjectsUnderPoint(x,y);
112 	}
113 	
114 	this.clone = function() {
115 		var o = new Stage(null);
116 		this.cloneProps(o);
117 		return o;
118 	}
119 		
120 	this.toString = function() {
121 		return "[Stage (name="+  this.name +")]";
122 	}
123 	
124 // private methods:
125 	/** @private **/
126 	this.__getObjectsUnderPoint = this._getObjectsUnderPoint;
127 	/** @private **/
128 	this._getObjectsUnderPoint = function(x,y,arr) {
129 		if (this._tmpCanvas == null) { this._tmpCanvas = document.createElement("canvas"); }
130 		this._tmpCanvas.width = this.canvas.width;
131 		this._tmpCanvas.height = this.canvas.height;
132 		
133 		var ctx = this._tmpCanvas.getContext("2d");
134 		
135 		this.updateContext(ctx,true);
136 		var results = this.__getObjectsUnderPoint(x,y,ctx,arr);
137 		this.revertContext();
138 		
139 		return results;
140 	}
141 	
142 	this._handleMouseMove = function(e) {
143 		if (!this.canvas) {
144 			this.mouseX = this.mouseY = null;
145 			return;
146 		}
147 		this.mouseX = e.pageX-this.canvas.offsetLeft;
148 		this.mouseY = e.pageY-this.canvas.offsetTop;
149 	}
150 }
